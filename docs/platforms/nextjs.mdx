# Next.js (App Router)

This guide walks through a complete production setup on **Next.js 14+ App Router** using `DrizzlePgStorage` with Neon and `SimpleRetryStrategy`.

---

## Install

```bash
npm install serverless-queue drizzle-orm @neondatabase/serverless resend
```

---

## Project structure

```
lib/
  queue/
    action.ts      ← EmailAction (business logic)
    index.ts       ← queue singleton + enqueue helper
app/
  api/
    queue/
      route.ts     ← POST handler
```

## `lib/queue/action.ts`

```ts
import { MessageAction } from 'serverless-queue'
import type { QueueMessage } from 'serverless-queue'
import { Resend } from 'resend'

export interface EmailPayload { to: string; subject: string; html: string }

const resend = new Resend(process.env.RESEND_API_KEY!)

export class EmailAction extends MessageAction<EmailPayload> {
  async execute(messages: QueueMessage<EmailPayload>[]) {
    for (const msg of messages) {
      const { error } = await resend.emails.send({
        from: process.env.EMAIL_FROM!,
        to: msg.payload.to,
        subject: msg.payload.subject,
        html: msg.payload.html,
        headers: { 'X-Queue-Message-Id': msg.id },
      })
      if (error) throw new Error(`Resend error: ${error.message}`)
    }
  }
}
```

## `lib/queue/index.ts`

```ts
import {
  MessageQueue, MessageInvoker, SimpleRetryStrategy, DrizzlePgStorage, pgQueueMessages,
} from 'serverless-queue'
import type { QueueMessage } from 'serverless-queue'
import { neon } from '@neondatabase/serverless'
import { drizzle } from 'drizzle-orm/neon-http'
import { EmailAction } from './action'
import type { EmailPayload } from './action'

const db = drizzle(neon(process.env.DATABASE_URL!))

class FetchInvoker extends MessageInvoker {
  async invoke(messages: QueueMessage[]) {
    const res = await fetch(`${process.env.NEXT_PUBLIC_APP_URL}/api/queue`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'x-queue-secret': process.env.QUEUE_SECRET!,
      },
      body: JSON.stringify(messages),
      signal: AbortSignal.timeout(8_000),
    })
    if (!res.ok) throw new Error(`Self-invoke failed: ${res.status}`)
  }
}

export const queue = new MessageQueue(
  new DrizzlePgStorage(db, pgQueueMessages),
  new EmailAction(),
  new SimpleRetryStrategy({
    shouldRetry: (_msg, err) => {
      if (err instanceof Error && err.message.includes('Invalid `to` field')) return false
      if (err instanceof Error && err.message.includes('API key')) return false
      return true
    },
    onExhausted: async (msg) => {
      console.error('[email-queue] exhausted', { id: msg.id, attempt: msg.attempt })
    },
  }),
  new FetchInvoker(),
  { batchSize: 5, maxAttempts: 4, invokeRetries: 3, drainMode: 'individual' },
)

export async function enqueue(payload: EmailPayload) {
  await queue.handle([{
    id: crypto.randomUUID(),
    queueId: payload.to,
    payload,
    attempt: 0,
    createdAt: Date.now(),
    type: 'new',
  }])
}

export type { EmailPayload }
```

## `app/api/queue/route.ts`

```ts
import { queue } from '@/lib/queue'
import type { QueueMessage } from 'serverless-queue'
import { NextRequest, NextResponse } from 'next/server'

export const maxDuration = 60

export async function POST(req: NextRequest): Promise<NextResponse> {
  if (req.headers.get('x-queue-secret') !== process.env.QUEUE_SECRET) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  }

  let messages: QueueMessage[]
  try {
    messages = await req.json()
  } catch {
    return NextResponse.json({ error: 'Invalid JSON' }, { status: 400 })
  }

  if (!Array.isArray(messages) || messages.length === 0) {
    return NextResponse.json({ error: 'Expected non-empty array' }, { status: 400 })
  }

  await queue.handle(messages)
  return NextResponse.json({ ok: true })
}
```

## Database migration

```sql
CREATE TABLE IF NOT EXISTS queue_messages (
  id          TEXT    NOT NULL PRIMARY KEY,
  queue_id    TEXT    NOT NULL,
  message     TEXT    NOT NULL,
  timestamp   BIGINT  NOT NULL,
  state       TEXT    NOT NULL DEFAULT 'pending'
);
CREATE INDEX IF NOT EXISTS idx_queue_messages_queue_id_timestamp
  ON queue_messages (queue_id, timestamp);
```

## Enqueueing a message

```ts
// app/actions/send-welcome-email.ts
'use server'

import { enqueue } from '@/lib/queue'

export async function sendWelcomeEmail(email: string) {
  await enqueue({
    to: email,
    subject: 'Welcome!',
    html: '<h1>Thanks for joining.</h1>',
  })
}
```

## Vercel-specific notes

### Function timeout

Set `maxDuration: 60` on the queue route. With a 8-second timeout on the `FetchInvoker`, there is plenty of headroom.

> ⚠️ **Warning:** On the Vercel Hobby plan, function duration is limited to 10 seconds. Upgrade to Pro for up to 300 seconds.

### Runtime

Use the **Node.js runtime** (default), not the Edge runtime.

```ts
export const runtime = 'nodejs' // explicit; this is the default
export const maxDuration = 60
```
