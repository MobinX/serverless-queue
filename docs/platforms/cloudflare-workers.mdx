# Cloudflare Workers

Cloudflare Workers is an excellent fit for `serverless-queue`: the V8 isolate model starts fast, D1 provides a SQLite-backed relational store, and `env.SELF.fetch` lets you self-invoke without knowing any external URL.

---

## Install

```bash
npm install serverless-queue drizzle-orm
```

---

## Project structure

```
src/
  queue/
    action.ts      ← your business logic
    index.ts       ← queue factory
  index.ts         ← fetch handler entrypoint
wrangler.toml
migrations/
  0001_create_queue_messages.sql
```

---

## `wrangler.toml`

```toml
name = "my-worker"
main = "src/index.ts"
compatibility_date = "2024-09-01"

[[d1_databases]]
binding = "DB"
database_name = "queue-db"
database_id = "your-d1-database-id"

[vars]
QUEUE_SECRET = "replace-with-a-real-secret"

# SELF binding enables self-invocation without an external URL
[[unsafe.bindings]]
name = "SELF"
type = "fetcher"
service = "my-worker"
```

> 📝 **Note:** The `SELF` binding routes requests back to the same worker script without leaving Cloudflare's network — zero-latency self-invocations that work identically in `wrangler dev` and production.

## D1 migration

```sql
-- migrations/0001_create_queue_messages.sql
CREATE TABLE IF NOT EXISTS queue_messages (
  id        TEXT    NOT NULL PRIMARY KEY,
  queue_id  TEXT    NOT NULL,
  message   TEXT    NOT NULL,
  timestamp INTEGER NOT NULL,
  state     TEXT    NOT NULL DEFAULT 'pending'
);
CREATE INDEX IF NOT EXISTS idx_queue_messages_queue_id_timestamp
  ON queue_messages (queue_id, timestamp);
```

Apply with:

```bash
wrangler d1 migrations apply queue-db
```

Or use `MIGRATION_SQL` from the library directly:

```ts
import { MIGRATION_SQL } from 'serverless-queue'
await env.DB.exec(MIGRATION_SQL)
```

## `src/queue/action.ts`

```ts
import { MessageAction } from 'serverless-queue'
import type { QueueMessage } from 'serverless-queue'

export interface NotificationPayload {
  userId: string
  type: 'welcome' | 'password_reset' | 'invoice'
  data: Record<string, string>
}

export class NotificationAction extends MessageAction<NotificationPayload> {
  constructor(private emailServiceUrl: string, private apiKey: string) {
    super()
  }

  async execute(messages: QueueMessage<NotificationPayload>[]) {
    for (const msg of messages) {
      const res = await fetch(this.emailServiceUrl, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${this.apiKey}` },
        body: JSON.stringify({ ...msg.payload, messageId: msg.id }),
      })
      if (!res.ok) throw new Error(`Notification service error: ${res.status}`)
    }
  }
}
```

## `src/queue/index.ts`

```ts
import {
  MessageQueue, MessageInvoker, SimpleRetryStrategy, DrizzleSqliteStorage,
} from 'serverless-queue'
import type { QueueMessage } from 'serverless-queue'
import { drizzle } from 'drizzle-orm/d1'
import { NotificationAction } from './action'
import type { NotificationPayload } from './action'

interface Env {
  DB: D1Database
  SELF: Fetcher
  QUEUE_SECRET: string
  EMAIL_SERVICE_URL: string
  EMAIL_API_KEY: string
}

class WorkerSelfInvoker extends MessageInvoker<NotificationPayload> {
  constructor(private self: Fetcher, private secret: string) {
    super()
  }

  async invoke(messages: QueueMessage<NotificationPayload>[]) {
    const res = await this.self.fetch('https://worker/api/queue', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${this.secret}`,
      },
      body: JSON.stringify(messages),
    })
    if (!res.ok) throw new Error(`Self-invoke failed: ${res.status}`)
  }
}

export function createQueue(env: Env) {
  return new MessageQueue<NotificationPayload>(
    new DrizzleSqliteStorage(drizzle(env.DB)),
    new NotificationAction(env.EMAIL_SERVICE_URL, env.EMAIL_API_KEY),
    new SimpleRetryStrategy({
      shouldRetry: (_msg, err) => !(err instanceof Error && err.message.includes('4')),
      onExhausted: async (msg) => console.error('Message exhausted', msg.id),
    }),
    new WorkerSelfInvoker(env.SELF, env.QUEUE_SECRET),
    { batchSize: 5, maxAttempts: 3, invokeRetries: 3, drainMode: 'individual' },
  )
}
```

## `src/index.ts`

```ts
import { createQueue } from './queue'
import type { QueueMessage } from 'serverless-queue'
import type { NotificationPayload } from './queue/action'

interface Env {
  DB: D1Database
  SELF: Fetcher
  QUEUE_SECRET: string
  EMAIL_SERVICE_URL: string
  EMAIL_API_KEY: string
}

export default {
  async fetch(request: Request, env: Env): Promise<Response> {
    const url = new URL(request.url)

    if (url.pathname === '/api/queue' && request.method === 'POST') {
      const auth = request.headers.get('authorization')
      if (auth !== `Bearer ${env.QUEUE_SECRET}`) {
        return new Response('Unauthorized', { status: 401 })
      }

      let messages: QueueMessage<NotificationPayload>[]
      try {
        messages = await request.json()
      } catch {
        return new Response('Invalid JSON', { status: 400 })
      }

      await createQueue(env).handle(messages)
      return Response.json({ ok: true })
    }

    return new Response('Not Found', { status: 404 })
  },
}
```

## Enqueueing from inside the worker

```ts
async function enqueue(env: Env, userId: string, type: NotificationPayload['type'], data: Record<string, string>) {
  const message: QueueMessage<NotificationPayload> = {
    id: crypto.randomUUID(),
    queueId: userId,
    payload: { userId, type, data },
    attempt: 0,
    createdAt: Date.now(),
    type: 'new',
  }

  const res = await env.SELF.fetch('https://worker/api/queue', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': `Bearer ${env.QUEUE_SECRET}`,
    },
    body: JSON.stringify([message]),
  })

  if (!res.ok) throw new Error(`Enqueue failed: ${res.status}`)
}
```

## Durable Objects vs `serverless-queue`

| | `serverless-queue` | Durable Objects |
|--|--|--|
| **State** | Your D1 database | Built-in per-object storage |
| **Serialisation** | Per `queueId` via DB lock | Per Durable Object instance |
| **Complexity** | Low — implement 2 abstract classes + config | Higher — custom WebSocket/alarm logic |
| **Cost** | D1 reads/writes (very cheap) | DO requests + storage |
| **Best for** | Queue workloads with an existing DB | Real-time collaboration, presence, gaming |

<details>
<summary><strong>Advanced: custom storage (without Drizzle)</strong></summary>

If you prefer raw D1 queries without Drizzle ORM you can implement `MessageStorage` manually. See the [Storage guide](../storage.mdx) for the complete D1 example using `db.prepare()`.

</details>
