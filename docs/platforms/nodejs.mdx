# Node.js / Express

`serverless-queue` works equally well in a traditional Node.js server. This guide uses **Express**, **DrizzleSqliteStorage** with `better-sqlite3`, and a simple `fetch`-based self-invoker.

---

## Install dependencies

```bash
npm install express better-sqlite3 serverless-queue drizzle-orm
npm install -D @types/express @types/better-sqlite3 tsx
```

## `src/queue/action.ts`

```ts
import { MessageAction } from 'serverless-queue'
import type { QueueMessage } from 'serverless-queue'

export interface JobPayload {
  type: 'send_report' | 'export_csv' | 'send_notification'
  userId: string
  params: Record<string, unknown>
}

export class JobAction extends MessageAction<JobPayload> {
  async execute(messages: QueueMessage<JobPayload>[]) {
    for (const msg of messages) {
      console.log(`[job-queue] executing ${msg.payload.type}`, { id: msg.id, attempt: msg.attempt })

      switch (msg.payload.type) {
        case 'send_report':
          await this.sendReport(msg.payload.userId, msg.payload.params); break
        case 'export_csv':
          await this.exportCsv(msg.payload.userId, msg.payload.params); break
        case 'send_notification':
          await this.sendNotification(msg.payload.userId, msg.payload.params); break
        default:
          throw new Error(`Unknown job type: ${(msg.payload as JobPayload).type}`)
      }
    }
  }

  private async sendReport(userId: string, params: Record<string, unknown>) {
    console.log('[job-queue] sending report', { userId, params })
  }
  private async exportCsv(userId: string, params: Record<string, unknown>) {
    console.log('[job-queue] exporting CSV', { userId, params })
  }
  private async sendNotification(userId: string, params: Record<string, unknown>) {
    console.log('[job-queue] sending notification', { userId, params })
  }
}
```

## `src/queue/index.ts`

```ts
import Database from 'better-sqlite3'
import { drizzle } from 'drizzle-orm/better-sqlite3'
import {
  MessageQueue, MessageInvoker, SimpleRetryStrategy, DrizzleSqliteStorage, MIGRATION_SQL,
} from 'serverless-queue'
import type { QueueMessage } from 'serverless-queue'
import { JobAction } from './action'
import type { JobPayload } from './action'

const BASE_URL = process.env.SERVER_BASE_URL ?? 'http://localhost:3000'
const QUEUE_SECRET = process.env.QUEUE_SECRET ?? 'dev-secret'

const sqlite = new Database('./queue.db')
sqlite.pragma('journal_mode = WAL')
sqlite.exec(MIGRATION_SQL)  // runs CREATE TABLE IF NOT EXISTS on startup

const db = drizzle(sqlite)

class SelfInvoker extends MessageInvoker<JobPayload> {
  async invoke(messages: QueueMessage<JobPayload>[]) {
    const res = await fetch(`${BASE_URL}/api/queue`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${QUEUE_SECRET}`,
      },
      body: JSON.stringify(messages),
    })
    if (!res.ok) throw new Error(`Self-invoke failed: ${res.status}`)
  }
}

export const jobQueue = new MessageQueue<JobPayload>(
  new DrizzleSqliteStorage(db),
  new JobAction(),
  new SimpleRetryStrategy({
    shouldRetry: (_msg, err) => !(err instanceof Error && err.message.startsWith('Unknown job type')),
    onExhausted: async (msg) => {
      console.error('[job-queue] exhausted', { id: msg.id, type: msg.payload.type })
    },
  }),
  new SelfInvoker(),
  { batchSize: 5, maxAttempts: 3, invokeRetries: 3, drainMode: 'individual' },
)

export type { JobPayload }
```

> 💡 `MIGRATION_SQL` from `serverless-queue` creates the `queue_messages` table and index. Calling `sqlite.exec(MIGRATION_SQL)` on startup is safe — it uses `CREATE TABLE IF NOT EXISTS`.

## `src/server.ts`

```ts
import express from 'express'
import { jobQueue } from './queue'
import type { QueueMessage } from 'serverless-queue'
import type { JobPayload } from './queue/action'

const app = express()
app.use(express.json())

const QUEUE_SECRET = process.env.QUEUE_SECRET ?? 'dev-secret'
const PORT = Number(process.env.PORT ?? 3000)

app.post('/api/queue', async (req, res) => {
  if (req.headers.authorization !== `Bearer ${QUEUE_SECRET}`) {
    res.status(401).json({ error: 'Unauthorized' }); return
  }
  const messages: QueueMessage<JobPayload>[] = req.body
  if (!Array.isArray(messages) || messages.length === 0) {
    res.status(400).json({ error: 'Expected non-empty array' }); return
  }
  await jobQueue.handle(messages)
  res.json({ ok: true })
})

app.post('/api/jobs', async (req, res) => {
  const { userId, type, params } = req.body
  const message: QueueMessage<JobPayload> = {
    id: crypto.randomUUID(),
    queueId: userId,
    payload: { userId, type, params },
    attempt: 0,
    createdAt: Date.now(),
    type: 'new',
  }
  await fetch(`http://localhost:${PORT}/api/queue`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${QUEUE_SECRET}` },
    body: JSON.stringify([message]),
  })
  res.json({ queued: true, id: message.id })
})

app.listen(PORT, () => console.log(`Server running at http://localhost:${PORT}`))
```

## Running locally

```bash
# Development (with auto-reload)
npx tsx watch src/server.ts

npx tsx src/server.ts
```

### Test it manually

```bash
# Enqueue a job
curl -X POST http://localhost:3000/api/jobs \
  -H 'Content-Type: application/json' \
  -d '{"userId":"user_123","type":"send_report","params":{"reportId":"rpt_456"}}'

# Check the database
sqlite3 queue.db "SELECT id, queue_id, state FROM queue_messages ORDER BY timestamp DESC LIMIT 10;"
```

> 💡 **Tip:** SQLite is great for development and low-traffic production. For higher throughput, swap `DrizzleSqliteStorage` for `DrizzlePgStorage` with a Postgres URL — no other code changes needed.
